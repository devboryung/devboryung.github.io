## 문제 10
```javascript
<script>
    var n = prompt('숫자를 입력하세요.');
    var tree = '';

    for(var i = 1; i <= n; i++){
        var star = '';
        for(var j=1; j<=n-i; j++){
            star += ' ';
        }
        for(var k=1; k<=2*i-1; k++){
            star += '*';
        }
        tree += star + '\n';
    }
    console.log(tree);
</script>
```
